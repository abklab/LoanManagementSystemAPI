CREATE proc [dbo].[usp_validateCustomerRefernceID]
	@refrenceID VARCHAR(50)='12345678912'
	
	AS
	 declare @outputValue int;  
	     
	 SELECT @outputValue = COUNT(*)  
	 FROM [dbo].[vwCustomersLoanOutstanding] 
	 where Account_Number = @refrenceID;

	return @outputValue
